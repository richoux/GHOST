132  virtual int v_heuristicValue( const std::vector< double > &vecGlobalCosts,
133  double &bestEstimatedCost,
134  int &bestValue ) const
135  {
136  int best = 0;
137  double bestHelp = numeric_limits<int>::max();
138 
139  vector<int> draw;
140 
141  for( int i = 0; i < vecGlobalCosts.size(); ++i )
142  {
143  if( vecGlobalCosts[i] == bestEstimatedCost
144  && vecGlobalCosts[i] < numeric_limits<int>::max()
145  && heuristicValueHelper.at( i ) == bestHelp )
146  {
147  draw.push_back(i);
148  }
149  else
150  if( vecGlobalCosts[i] < bestEstimatedCost
151  || ( vecGlobalCosts[i] == bestEstimatedCost
152  && vecGlobalCosts[i] < numeric_limits<int>::max()
153  && heuristicValueHelper.at( i ) < bestHelp ) )
154  {
155  draw.clear();
156  bestEstimatedCost = vecGlobalCosts[i];
157  bestValue = i - 1;
158  if( heuristicValueHelper.at( i ) < bestHelp )
159  bestHelp = heuristicValueHelper.at( i );
160  best = i;
161  }
162  }
163 
164  if( draw.size() > 1 )
165  {
166  int i = draw[ const_cast<Objective*>(this)->randomVar.getRandNum( draw.size() ) ];
167 
168  bestEstimatedCost = vecGlobalCosts[i];
169  bestValue = i - 1;
170  if( heuristicValueHelper.at( i ) < bestHelp )
171  bestHelp = heuristicValueHelper.at( i );
172  best = i;
173  }
174 
175  return best;
176  }
