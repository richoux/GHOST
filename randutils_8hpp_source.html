157 /*
158  * seed_seq_fe implements a fixed-entropy seed sequence; it conforms to all
159  * the requirements of a Seed Sequence concept.
160  *
161  * seed_seq_fe<N> implements a seed sequence which seeds based on a store of
162  * N * 32 bits of entropy. Typically, it would be initialized with N or more
163  * integers.
164  *
165  * seed_seq_fe128 and seed_seq_fe256 are provided as convenience typedefs for
166  * 128- and 256-bit entropy stores respectively. These variants outperform
167  * std::seed_seq, while being better mixing the bits it is provided as entropy.
168  * In almost all common use cases, they serve as better drop-in replacements
169  * for seed_seq.
170  *
171  * Technical details
172  *
173  * Assuming it constructed with M seed integers as input, it exhibits the
174  * following properties
175  *
176  * * Diffusion/Avalanche: A single-bit change in any of the M inputs has a
177  * 50% chance of flipping every bit in the bitstream produced by generate.
178  * Initializing the N-word entropy store with M words requires O(N * M)
179  * time precisely because of the avalanche requirements. Once constructed,
180  * calls to generate are linear in the number of words generated.
181  *
182  * * Bias freedom/Bijection: If M == N, the state of the entropy store is a
183  * bijection from the M inputs (i.e., no states occur twice, none are
184  * omitted). If M > N the number of times each state can occur is the same
185  * (each state occurs 2**(32*(M-N)) times, where ** is the power function).
186  * If M < N, some states cannot occur (bias) but no state occurs more
187  * than once (it's impossible to avoid bias if M < N; ideally N should not
188  * be chosen so that it is more than M).
189  *
190  * Likewise, the generate function has similar properties (with the entropy
191  * store as the input data). If more outputs are requested than there is
192  * entropy, some outputs cannot occur. For example, the Mersenne Twister
193  * will request 624 outputs, to initialize it's 19937-bit state, which is
194  * much larger than a 128-bit or 256-bit entropy pool. But in practice,
195  * limiting the Mersenne Twister to 2**128 possible initializations gives
196  * us enough initializations to give a unique initialization to trillions
197  * of computers for billions of years. If you really have 624 words of
198  * *real* high-quality entropy you want to use, you probably don't need
199  * an entropy mixer like this class at all. But if you *really* want to,
200  * nothing is stopping you from creating a randutils::seed_seq_fe<624>.
201  *
202  * * As a consequence of the above properties, if all parts of the provided
203  * seed data are kept constant except one, and the remaining part is varied
204  * through K different states, K different output sequences will be produced.
205  *
206  * * Also, because the amount of entropy stored is fixed, this class never
207  * performs dynamic allocation and is free of the possibility of generating
208  * an exception.
209  *
210  * Ideas used to implement this code include hashing, a simple PCG generator
211  * based on an MCG base with an XorShift output function and permutation
212  * functions on tuples.
213  *
214  * More detail at
215  * http://www.pcg-random.org/posts/developing-a-seed_seq-alternative.html
216  */

236  RANDUTILS_GENERALIZED_CONSTEXPR
237  static IntRep fast_exp(IntRep x, IntRep power)
238  {
239  IntRep result = IntRep(1);
240  IntRep multiplier = x;
241  while (power != IntRep(0)) {
242  IntRep thismult = power & IntRep(1) ? multiplier : IntRep(1);
243  result *= thismult;
244  power >>= 1;
245  multiplier *= multiplier;
246  }
247  return result;
248  }

301  template <size_t count, typename IntRep, size_t r>
302  template <typename InputIter>
303  void seed_seq_fe<count, IntRep, r>::mix_entropy(InputIter begin, InputIter end)
304  {
305  auto hash_const = INIT_A;
306  auto hash = [&](IntRep value) {
307  value ^= hash_const;
308  hash_const *= MULT_A;
309  value *= hash_const;
310  value ^= value >> XSHIFT;
311  return value;
312  };
313  auto mix = [](IntRep x, IntRep y) {
314  IntRep result = MIX_MULT_L*x - MIX_MULT_R*y;
315  result ^= result >> XSHIFT;
316  return result;
317  };
318 
319  InputIter current = begin;
320  for (auto& elem : mixer_) {
321  if (current != end)
322  elem = hash(*current++);
323  else
324  elem = hash(0U);
325  }
326  for (auto& src : mixer_)
327  for (auto& dest : mixer_)
328  if (&src != &dest)
329  dest = mix(dest,hash(src));
330  for (; current != end; ++current)
331  for (auto& dest : mixer_)
332  dest = mix(dest,hash(*current));
333  }

335  template <size_t count, typename IntRep, size_t mix_rounds>
336  template <typename OutputIterator>
337  void seed_seq_fe<count,IntRep,mix_rounds>::param(OutputIterator dest) const
338  {
339  const IntRep INV_A = fast_exp(MULT_A, IntRep(-1));
340  const IntRep MIX_INV_L = fast_exp(MIX_MULT_L, IntRep(-1));
341 
342  auto mixer_copy = mixer_;
343  for (size_t round = 0; round < mix_rounds; ++round) {
344  // Advance to the final value. We'll backtrack from that.
345  auto hash_const = INIT_A*fast_exp(MULT_A, IntRep(count * count));
346 
347  for (auto src = mixer_copy.rbegin(); src != mixer_copy.rend(); ++src)
348  for (auto dest = mixer_copy.rbegin(); dest != mixer_copy.rend();
349  ++dest)
350  if (src != dest) {
351  IntRep revhashed = *src;
352  auto mult_const = hash_const;
353  hash_const *= INV_A;
354  revhashed ^= hash_const;
355  revhashed *= mult_const;
356  revhashed ^= revhashed >> XSHIFT;
357  IntRep unmixed = *dest;
358  unmixed ^= unmixed >> XSHIFT;
359  unmixed += MIX_MULT_R*revhashed;
360  unmixed *= MIX_INV_L;
361  *dest = unmixed;
362  }
363  for (auto i = mixer_copy.rbegin(); i != mixer_copy.rend(); ++i) {
364  IntRep unhashed = *i;
365  unhashed ^= unhashed >> XSHIFT;
366  unhashed *= fast_exp(hash_const, IntRep(-1));
367  hash_const *= INV_A;
368  unhashed ^= hash_const;
369  *i = unhashed;
370  }
371  }
372  std::copy(mixer_copy.begin(), mixer_copy.end(), dest);
373  }

376  template <size_t count, typename IntRep, size_t mix_rounds>
377  template <typename RandomAccessIterator>
378  void seed_seq_fe<count,IntRep,mix_rounds>::generate(
379  RandomAccessIterator dest_begin,
380  RandomAccessIterator dest_end) const
381  {
382  auto src_begin = mixer_.begin();
383  auto src_end = mixer_.end();
384  auto src = src_begin;
385  auto hash_const = INIT_B;
386  for (auto dest = dest_begin; dest != dest_end; ++dest) {
387  auto dataval = *src;
388  if (++src == src_end)
389  src = src_begin;
390  dataval ^= hash_const;
391  hash_const *= MULT_B;
392  dataval *= hash_const;
393  dataval ^= dataval >> XSHIFT;
394  *dest = dataval;
395  }
396  }
